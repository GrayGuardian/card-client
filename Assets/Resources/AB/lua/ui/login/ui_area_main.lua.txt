local base = require("ui_base")
local UI = class("area_main", base)

----------------UI配置------------------
function UI:GET_AB()
    return "ui1"
end
function UI:GET_DEFAULT_OPTS()
    return {blockInput = false, blackMask = 0}
end
----------------UI配置------------------

function UI:onLoad()
    self.areaItemComp = comp:addComponent(self.node:Find("areaItem"), "area_item", {info = GameData.areaInfo})
end
function UI:onAreaItemClick(o, order)
    ui:showUI(
        "ui_area_sel",
        {
            area = GameData.areaInfo,
            areas = GameData.areaInfos,
            onClick = function(info)
                GameData.areaInfo = info
                self.areaItemComp.info = GameData.areaInfo
                self.areaItemComp:refresh()
            end
        }
    )
end
function UI:onNextBtnClick(o, order)
    if GameData.areaInfo.ismaintenance == 1 then
        ui:showTip({msg = "区服正在维护"})
        return
    end
    http:send(
        "nextArea",
        {aid = GameData.areaInfo.aid},
        function(data)
            print("进入选人界面")

            GameData.players = data.players
            ui:showUI("ui_player_sel")
            self:close()
        end
    )
end
return UI
